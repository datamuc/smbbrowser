        def mime_type
            name = @smbfile.getName
            mime = MIME::Types.type_for(name)[0]

            if ['text/plain'].find_index(mime)
                return mime.to_s + "; charset=utf-8"
            end

            return mime.to_s if mime
            return 'text/plain; charset=UTF-8'
        end
