        def File::get(file, domain=nil, user=nil, pass=nil)
            if pass and !pass.empty?
                auth = CIFS::NtlmPasswordAuthentication.new(domain, user, pass)
                smbfile = CIFS::SmbFile.new(file, auth)
            else
                smbfile = CIFS::SmbFile.new(file)
            end
            return smbfile
        end
