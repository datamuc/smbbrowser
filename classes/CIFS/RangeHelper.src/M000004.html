        def initialize(spec, length)
            @spec = spec
            @vals = spec.split('-').map{ |x| x.empty? ? nil : x.to_i }
            @length = length
            @first = create_first
            @last  = create_last
            if @first > @last
                raise RangeNotSatisfiableException, \
                    "invalid spec #{ @spec } first #{@first} last #{@last} length #{@length}"
            end
        end
